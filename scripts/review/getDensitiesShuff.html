dico = {}
for sp in spList:
    filename = path+'data/'+sp+'/Sequences_Gene_WT.fa'
    try:
        fastaOrigin = SeqIO.parse(open(filename),'fasta')
    except:
        pass
    else:
        for fasta in fastaOrigin:
            name, seq = fasta.id, str(fasta.seq)
            if sp not in dico:
                dico[sp] = 0
            dico[sp] += len(seq)
